\begin{align*}
\mathrm{LayerNorm}(z;\gamma,\beta) \;=\; \gamma \,\frac{z - \mu_z}{\sigma_z} \;+\; \beta,\quad\mu_z = \frac{1}{d}\sum_{i=1}^{d} z_i,\quad\sigma_z = \sqrt{\frac{1}{d}\sum_{i=1}^{d} (z_i - \mu_z)^2}\,.
\end{align*}
